js
module.exports = {
  devServer: {
    proxy: {
      '/api1': {
        // еҢ№й…ҚжүҖжңүд»Ҙ '/api1'ејҖеӨҙзҡ„иҜ·жұӮи·Ҝеҫ„
        target: 'http://localhost:5000', // д»ЈзҗҶзӣ®ж Үзҡ„еҹәзЎҖи·Ҝеҫ„
        changeOrigin: true,
        // жҠҠи·Ҝеҫ„зҡ„ /api1 жӣҝжҚўдёәз©әдёІ
        pathRewrite: { '^/api1': '' },
      },
      '/api2': {
        // еҢ№й…ҚжүҖжңүд»Ҙ '/api2'ејҖеӨҙзҡ„иҜ·жұӮи·Ҝеҫ„
        target: 'http://localhost:5001',

        pathRewrite: { '^/api2': '' },
      },
    },
  },
}

vue 2.x е…ЁеұҖй…ҚзҪ® axios вҖӢ

е®һйҷ…йЎ№зӣ®ејҖеҸ‘дёӯпјҢеҮ д№ҺжҜҸдёӘз»„д»¶дёӯйғҪдјҡдҪҝз”Ё axios еҸ‘иө·ж•°жҚ®иҜ·жұӮгҖӮжӯӨж—¶дјҡйҒҮеҲ°еҰӮдёӢдёӨдёӘй—®йўҳпјҡ

  • жҜҸдёӘз»„д»¶дёӯйғҪйңҖиҰҒеҜје…Ҙ axiosпјҲд»Јз ҒиҮғиӮҝпјү
  • жҜҸж¬ЎеҸ‘иҜ·жұӮйғҪйңҖиҰҒеЎ«еҶҷе®Ңж•ҙзҡ„иҜ·жұӮи·Ҝеҫ„пјҲдёҚеҲ©дәҺеҗҺжңҹзҡ„з»ҙжҠӨпјү

дјҳзӮ№пјҡжҜҸдёӘз»„д»¶еҸҜд»ҘйҖҡиҝҮ this.$http.get зӣҙжҺҘеҸ‘иө·иҜ·жұӮпјҢж— йңҖеҶҚеҜје…Ҙ axios пјӣиӢҘж №и·Ҝеҫ„еҸ‘з”ҹж”№еҸҳпјҢеҸӘйңҖдҝ®ж”№ axios.defaults.baseURL пјҢжңүеҲ©дәҺд»Јз Ғз»ҙжҠӨгҖӮ

зјәзӮ№пјҡж— жі•е®һзҺ° API зҡ„еӨҚз”ЁгҖӮеҚіеӨҡдёӘз»„д»¶йңҖиҰҒеҜ№еҗҢдёҖдёӘжҺҘеҸЈеҸ‘иө·иҜ·жұӮпјҢйӮЈд№ҲжҜҸдёӘз»„д»¶йғҪйңҖиҰҒйҮҚеӨҚд№ҰеҶҷ this.$http.get('/users') зұ»дјјзҡ„д»Јз ҒпјҢйҖ жҲҗеҶ—дҪҷгҖӮпјҲи§Ҷйў‘дёҠзҡ„иҜҙжі•пјҢдёӘдәәи®ӨдёәзңҹжӯЈзҡ„зјәзӮ№жҳҜеҰӮжһңеӯҳеңЁеӨҡдёӘж №и·Ҝеҫ„пјҢиҝҷз§Қж–№ејҸж— жі•и§ЈеҶіпјҢжүҖд»ҘжүҚдјҡжңүдёӢйқўзҡ„ж”№иҝӣж–№ејҸгҖӮпјү

ж”№иҝӣпјҡеҜ№дәҺжҜҸдёҖдёӘж №и·Ҝеҫ„пјҢзӢ¬з«Ӣе°ҒиЈ…дёҖдёӘ request.js жЁЎеқ—пјҢз»„д»¶еҜје…ҘжүҖйңҖж №и·Ҝеҫ„еҜ№еә”зҡ„ axios иҝӣиЎҢдҪҝз”ЁгҖӮ
